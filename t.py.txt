import numpy as np

def getAllocArray(mapping, facIDs, secIDs):
    allocArray = np.zeros((len(facIDs), len(secIDs)), dtype=bool)
    mappingFacilities = list(mapping.keys())
    facIDs = [str(facID) for facID in facIDs]
    
    for fID in mappingFacilities:
        row = [facID == fID for facID in facIDs]
        sIDs = mapping[fID]
        
        for sID in sIDs:
            col = [secID == sID for secID in secIDs]
            allocArray[row, col] = True
    
    return allocArray
