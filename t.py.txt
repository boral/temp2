def init_crr(self):
    score_ranges = [
        (0, 5, 8.3),
        (5, 15, 8.2),
        (15, 25, 8.1),
        (25, 35, 7.2),
        (35, 45, 7.1),
        (45, 50, 6.2),
        (50, 54, 6.1),
        (54, 56, 5.3),
        (56, 59.5, 5.2),
        (59.5, 61, 5.1),
        (61, 63, 4.3),
        (63, 65, 4.2),
        (65, 67, 4.1),
        (67, 69, 3.3),
        (69, 72, 3.2),
        (72, 74, 3.1),
        (74, 81, 2.2),
        (81, 86, 2.1),
        (86, 90, 1.2),
        (90, 95, 1.1),
        (95, 100, 0.1)
    ]

    for range_start, range_end, crr_value in score_ranges:
        if range_start < self.Modelscore <= range_end:
            self.CRR = crr_value
            break

    return self.CRR
