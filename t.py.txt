def init_crr(self):
    modelscore = self.Modelscore
    score_ranges = [
        (0, 15, 8.3),
        (15, 25, 8.2),
        (25, 35, 8.1),
        (35, 40, 7.2),
        (40, 45, 7.1),
        (45, 50, 6.2),
        (50, 54, 6.1),
        (54, 57, 5.3),
        (57, 60, 5.2),
        (60, 63, 5.1),
        (63, 64.5, 4.3),
        (64.5, 66, 4.2),
        (66, 68, 4.1),
        (68, 70.5, 3.3),
        (70.5, 74, 3.2),
        (74, 83, 3.1),
        (83, 90, 2.2),
        (90, 93, 2.1),
        (93, 96, 1.2),
        (96, 98, 1.1),
        (98, float('inf'), 0.1)
    ]

    for range_start, range_end, crr in score_ranges:
        if range_start <= modelscore <= range_end:
            return crr
