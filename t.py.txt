import pandas as pd
import statsmodels.api as sm

# Example DataFrame creation
df = pd.DataFrame({'y': [1, 2, 3, 4, 5],
                   'x': [2, 4, 6, 8, 10]})

# Regression with intercept term
X_with_intercept = sm.add_constant(df['x'])
model_with_intercept = sm.OLS(df['y'], X_with_intercept).fit()
print("Regression with intercept term:")
print(model_with_intercept.summary())

# Regression without intercept term
model_without_intercept = sm.OLS(df['y'], df['x']).fit()
print("\nRegression without intercept term:")
print(model_without_intercept.summary())
