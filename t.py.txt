import re

def asIdentifier(id):
    """
    Makes a valid identifier from a string.
    """
    # Replace all non-alphanumeric characters with underscores
    id = re.sub(r'\W|^(?=\d)', '_', id)
    
    # Remove leading and trailing underscores
    id = id.strip('_')
    
    # Return the modified string
    return id
