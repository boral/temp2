import numpy as np

securityfilter = np.array([1, 0, 0, 0, 1])
associationarray = np.array([[False, False, False, False, False],
                             [False, False, True, True, True],
                             [False, False, False, False, False]])
validocrr = np.array([[True, True, True, False, False]])

# Convert False to 0 and True to 1 for array operations
associationarray = associationarray.astype(int)
validocrr = validocrr.astype(int)

# Perform the operations
haveecagaurantee = ((associationarray * securityfilter) > 0) * np.sum((1 * validocrr * associationarray), 1)

print(haveecagaurantee)
